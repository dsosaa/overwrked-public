    üîå Telegram Bot API Reference¶

    Authentication¶

    All API calls require a valid bot token included in the request headers:

    Authorization: Bot YOUR_BOT_TOKEN
Content-Type: application/json

    Base URL¶

    https://api.telegram.org/bot{token}/

    üì¨ Core Methods¶

    Send Message¶

    Send text messages to users or groups.

    import requests

def send_message(chat_id: int, text: str, parse_mode: str = "HTML"):
    url = f"https://api.telegram.org/bot{BOT_TOKEN}/sendMessage"
    payload = {
        "chat_id": chat_id,
        "text": text,
        "parse_mode": parse_mode
    }
    response = requests.post(url, json=payload)
    return response.json()

    Parameters: - chat_id (int): Target chat identifier - text (str): Message content (1-4096 characters) - parse_mode (str): "HTML", "Markdown", or "MarkdownV2"

    def create_inline_keyboard(buttons: list):
    keyboard = []
    for row in buttons:
        keyboard_row = []
        for button in row:
            keyboard_row.append({
                "text": button["text"],
                "callback_data": button["callback_data"]
            })
        keyboard.append(keyboard_row)

    return {"inline_keyboard": keyboard}

# Example usage
buttons = [
    [{"text": "üí≥ Buy 10 Credits ($5)", "callback_data": "buy_10"}],
    [{"text": "üíé Buy 50 Credits ($20)", "callback_data": "buy_50"}],
    [{"text": "üöÄ Buy 100 Credits ($35)", "callback_data": "buy_100"}]
]

keyboard = create_inline_keyboard(buttons)

    Format Credit Balance¶

    def format_balance_bar(current: int, maximum: int, length: int = 10) -> str:
    """Create a visual progress bar for credit balance"""
    if maximum == 0:
        return "‚ñë" * length

    filled = int((current / maximum) * length)
    empty = length - filled

    bar = "‚ñà" * filled + "‚ñë" * empty
    percentage = int((current / maximum) * 100)

    return f"[{bar}] {percentage}% ({current}/{maximum})"

# Example usage
balance_display = format_balance_bar(75, 100)
# Output: [‚ñà‚ñà‚ñà‚ñà‚ñà‚ñà‚ñà‚ñë‚ñë‚ñë] 75% (75/100)

    Rate Limiting¶

    from time import time
from collections import defaultdict

class RateLimiter:
    def __init__(self, max_requests: int = 30, window: int = 60):
        self.max_requests = max_requests
        self.window = window
        self.requests = defaultdict(list)

    def is_allowed(self, user_id: int) -> bool:
        now = time()
        user_requests = self.requests[user_id]

        # Remove old requests outside the window
        user_requests[:] = [req_time for req_time in user_requests if now - req_time < self.window]

        if len(user_requests) >= self.max_requests:
            return False

        user_requests.append(now)
        return True

# Usage
rate_limiter = RateLimiter(max_requests=20, window=60)

if not rate_limiter.is_allowed(user_id):
    send_message(user_id, "‚ö†Ô∏è Rate limit exceeded. Please slow down.")
    return

    Common Error Codes¶

    Code Description Solution
    400 Bad Request Check request parameters
    401 Unauthorized Verify bot token
    403 Forbidden User blocked bot or insufficient permissions
    404 Not Found Chat/user doesn't exist
    429 Too Many Requests Implement rate limiting
    500 Internal Server Error Retry after delay

    Error Handler¶

    import logging
from functools import wraps

def handle_telegram_errors(func):
    @wraps(func)
    def wrapper(*args, **kwargs):
        try:
            return func(*args, **kwargs)
        except Exception as e:
            logging.error(f"Telegram API error in {func.__name__}: {e}")
            # Implement fallback behavior
            return {"ok": False, "error": str(e)}
    return wrapper

@handle_telegram_errors
def send_message_safe(chat_id: int, text: str):
    return send_message(chat_id, text)
