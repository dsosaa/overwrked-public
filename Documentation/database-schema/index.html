    🗄️ Database Schema Documentation¶

    Overview¶

    The database schema is designed for scalability, performance, and data integrity. It supports user management, conversation tracking, payment processing, and comprehensive analytics.

    CREATE TABLE users (
    id BIGINT PRIMARY KEY,
    username VARCHAR(255),
    first_name VARCHAR(255) NOT NULL,
    last_name VARCHAR(255),
    credits INTEGER DEFAULT 0 CHECK (credits >= 0),
    unlimited_until TIMESTAMP,
    created_at TIMESTAMP DEFAULT NOW(),
    last_active TIMESTAMP DEFAULT NOW(),
    total_spent DECIMAL(10,2) DEFAULT 0 CHECK (total_spent >= 0),
    message_count INTEGER DEFAULT 0 CHECK (message_count >= 0),
    is_banned BOOLEAN DEFAULT FALSE,
    language_code VARCHAR(10) DEFAULT 'en',
    preferences JSONB DEFAULT '{}'::jsonb
);

-- Indexes for performance
CREATE INDEX idx_users_username ON users(username);
CREATE INDEX idx_users_last_active ON users(last_active DESC);
CREATE INDEX idx_users_unlimited_until ON users(unlimited_until) WHERE unlimited_until IS NOT NULL;
CREATE INDEX idx_users_total_spent ON users(total_spent DESC);

    CREATE TABLE products (
    id SERIAL PRIMARY KEY,
    name VARCHAR(255) NOT NULL,
    description TEXT,
    price DECIMAL(10,2) NOT NULL CHECK (price >= 0),
    credits INTEGER DEFAULT 0 CHECK (credits >= 0),
    unlimited_days INTEGER DEFAULT 0 CHECK (unlimited_days >= 0),
    is_active BOOLEAN DEFAULT TRUE,
    stripe_price_id VARCHAR(255) UNIQUE,
    created_at TIMESTAMP DEFAULT NOW(),

    -- Ensure product grants either credits OR unlimited time
    CONSTRAINT valid_product CHECK (
        (credits > 0 AND unlimited_days = 0) OR 
        (credits = 0 AND unlimited_days > 0)
    )
);

    CREATE TABLE transactions (
    id SERIAL PRIMARY KEY,
    user_id BIGINT REFERENCES users(id) ON DELETE CASCADE,
    stripe_session_id VARCHAR(255) UNIQUE,
    stripe_payment_intent_id VARCHAR(255),
    amount DECIMAL(10,2) NOT NULL CHECK (amount >= 0),
    credits_granted INTEGER DEFAULT 0 CHECK (credits_granted >= 0),
    unlimited_days_granted INTEGER DEFAULT 0 CHECK (unlimited_days_granted >= 0),
    product_id INTEGER REFERENCES products(id),
    status VARCHAR(50) DEFAULT 'pending' CHECK (status IN ('pending', 'completed', 'failed', 'refunded')),
    created_at TIMESTAMP DEFAULT NOW(),
    completed_at TIMESTAMP,
    metadata JSONB DEFAULT '{}'::jsonb
);

-- Indexes

    CREATE VIEW user_analytics AS
SELECT 
    u.id,
    u.first_name,
    u.username,
    u.credits,
    u.total_spent,
    u.message_count,
    u.created_at,
    u.last_active,
    COUNT(t.id) as transaction_count,
    COALESCE(SUM(t.amount), 0) as lifetime_value,
    AVG(t.amount) as avg_transaction_value,
    CASE 
        WHEN u.unlimited_until > NOW() THEN 'unlimited'
        WHEN u.credits > 0 THEN 'active'
        ELSE 'inactive'
    END as account_status
FROM users u
LEFT JOIN transactions t ON u.id = t.user_id AND t.status = 'completed'
GROUP BY u.id;

    CREATE VIEW revenue_analytics AS
SELECT 
    DATE(created_at) as date,
    COUNT(*) as transaction_count,
    SUM(amount) as daily_revenue,
    AVG(amount) as avg_transaction_value,
    COUNT(DISTINCT user_id) as unique_customers
FROM transactions 
WHERE status = 'completed'
GROUP BY DATE(created_at)
ORDER BY date DESC;

    CREATE OR REPLACE FUNCTION get_active_users(days_back INTEGER DEFAULT 30)
RETURNS TABLE(
    total_active BIGINT,
    new_users BIGINT,
    paying_users BIGINT,
    total_revenue DECIMAL
) AS $$
BEGIN
    RETURN QUERY
    SELECT 
        COUNT(DISTINCT u.id) as total_active,
        COUNT(DISTINCT CASE WHEN u.created_at > NOW() - INTERVAL '1 day' * days_back THEN u.id END) as new_users,
        COUNT(DISTINCT CASE WHEN t.user_id IS NOT NULL THEN u.id END) as paying_users,
        COALESCE(SUM(t.amount), 0) as total_revenue
    FROM users u
    LEFT JOIN transactions t ON u.id = t.user_id 
        AND t.status = 'completed' 
        AND t.created_at > NOW() - INTERVAL '1 day' * days_back
    WHERE u.last_active > NOW() - INTERVAL '1 day' * days_back;
END;
$$ LANGUAGE plpgsql;

    -- Add credits to user account
CREATE OR REPLACE FUNCTION add_user_credits(
    p_user_id BIGINT,
    p_credits INTEGER
) RETURNS BOOLEAN AS $$
BEGIN
    UPDATE users 
    SET credits = credits + p_credits,
        last_active = NOW()
    WHERE id = p_user_id;

    RETURN FOUND;
END;
$$ LANGUAGE plpgsql;

-- Deduct credits for message
CREATE OR REPLACE FUNCTION deduct_message_credit(
    p_user_id BIGINT,
    p_cost DECIMAL DEFAULT 1.00
) RETURNS BOOLEAN AS $$
DECLARE
    user_credits INTEGER;
    has_unlimited BOOLEAN;
BEGIN
    -- Check if user has unlimited access
    SELECT 
        credits,
        (unlimited_until IS NOT NULL AND unlimited_until > NOW())
    INTO user_credits, has_unlimited
    FROM users 
    WHERE id = p_user_id;

    -- If unlimited access, no deduction needed
    IF has_unlimited THEN
        RETURN TRUE;
    END IF;

    -- Check if user has enough credits
    IF user_credits < p_cost THEN
        RETURN FALSE;
    END IF;

    -- Deduct credits
    UPDATE users 
    SET credits = credits - p_cost,
        message_count = message_count + 1,
        last_active = NOW()
    WHERE id = p_user_id;

    RETURN TRUE;
END;
$$ LANGUAGE plpgsql;

    -- Get or create conversation topic
CREATE OR REPLACE FUNCTION get_or_create_topic(
    p_user_id BIGINT,
    p_group_id BIGINT,
    p_topic_id INTEGER DEFAULT NULL,
    p_topic_name VARCHAR DEFAULT NULL
) RETURNS INTEGER AS $$
DECLARE
    existing_topic_id INTEGER;
BEGIN
    -- Try to get existing topic
    SELECT topic_id INTO existing_topic_id
    FROM conversation_topics
    WHERE user_id = p_user_id AND group_id = p_group_id;

    -- If topic exists, return it
    IF existing_topic_id IS NOT NULL THEN
        -- Update last message time
        UPDATE conversation_topics 
        SET last_message_at = NOW(),
            message_count = message_count + 1
        WHERE user_id = p_user_id AND group_id = p_group_id;

        RETURN existing_topic_id;
    END IF;

    -- Create new topic if provided
    IF p_topic_id IS NOT NULL THEN
        INSERT INTO conversation_topics (
            user_id, topic_id, group_id, topic_name
        ) VALUES (
            p_user_id, p_topic_id, p_group_id, 
            COALESCE(p_topic_name, 'User ' || p_user_id)
        );

        RETURN p_topic_id;
    END IF;

    -- No topic found and none provided
    RETURN NULL;
END;
$$ LANGUAGE plpgsql;

    -- Materialized view for dashboard statistics
CREATE MATERIALIZED VIEW dashboard_stats AS
SELECT 
    (SELECT COUNT(*) FROM users) as total_users,
    (SELECT COUNT(*) FROM users WHERE last_active > NOW() - INTERVAL '7 days') as active_users_7d,
    (SELECT COUNT(*) FROM users WHERE created_at > NOW() - INTERVAL '24 hours') as new_users_24h,
    (SELECT SUM(amount) FROM transactions WHERE status = 'completed' AND created_at > NOW() - INTERVAL '24 hours') as revenue_24h,
    (SELECT COUNT(*) FROM message_logs WHERE sent_at > NOW() - INTERVAL '24 hours') as messages_24h;

-- Refresh every hour
CREATE OR REPLACE FUNCTION refresh_dashboard_stats()
RETURNS void AS $$
BEGIN
    REFRESH MATERIALIZED VIEW dashboard_stats;
END;
$$ LANGUAGE plpgsql;

    -- Monthly revenue growth
SELECT 
    DATE_TRUNC('month', created_at) as month,
    SUM(amount) as revenue,
    COUNT(*) as transactions,
    COUNT(DISTINCT user_id) as unique_customers,
    LAG(SUM(amount)) OVER (ORDER BY DATE_TRUNC('month', created_at)) as prev_month_revenue,
    ROUND(
        (SUM(amount) - LAG(SUM(amount)) OVER (ORDER BY DATE_TRUNC('month', created_at))) 
        / LAG(SUM(amount)) OVER (ORDER BY DATE_TRUNC('month', created_at)) * 100, 2
    ) as growth_percentage
FROM transactions 
WHERE status = 'completed'
GROUP BY DATE_TRUNC('month', created_at)
ORDER BY month DESC;

    -- User engagement metrics
SELECT 
    DATE(sent_at) as date,
    COUNT(DISTINCT user_id) as active_users,
    COUNT(*) as total_messages,
    ROUND(COUNT(*)::DECIMAL / COUNT(DISTINCT user_id), 2) as avg_messages_per_user,
    COUNT(CASE WHEN is_admin_reply THEN 1 END) as admin_replies,
    ROUND(
        COUNT(CASE WHEN is_admin_reply THEN 1 END)::DECIMAL / 
        COUNT(CASE WHEN NOT is_admin_reply THEN 1 END) * 100, 2
    ) as response_rate_percent
FROM message_logs 
WHERE sent_at > NOW() - INTERVAL '30 days'
GROUP BY DATE(sent_at)
ORDER BY date DESC;
